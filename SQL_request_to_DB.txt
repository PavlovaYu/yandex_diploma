SELECT c.login, 
       COUNT(DISTINCT o.track)
FROM "Couriers" AS c
LEFT JOIN "Orders" AS o ON c.id=o."courierId"
WHERE "inDelivery" = 't'
GROUP BY c.login;

SELECT track,
       CASE 
           WHEN "inDelivery" = 't' AND finished = 'f' THEN 1
           WHEN "inDelivery" = 't' AND finished = 't' THEN 2
           WHEN cancelled = 't' THEN -1
           ELSE 0
       END AS statuses
FROM "Orders";
